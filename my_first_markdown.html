Plot the data by subject

library(ggplot2)
ggplot(sleep, aes(x = Days, y = Reaction)) + 
  geom_point() +
  geom_smooth() +
  scale_x_continuous(breaks = seq(0, 8, by = 2)) +
  facet_wrap(~ Subject)
## `geom_smooth()` using method = 'loess'
